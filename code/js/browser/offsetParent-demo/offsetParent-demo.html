经测试，offsetParent属性获取到包含块节点(祖先节点中第一个position为relative、absolute、fixed的节点)。 

    <div id="first-demo-parent" style="position: relative;">
        <div id="first-demo">xxxxxxxxxxxxxxxxxxxx</div>
    </div>
    $("#first-demo").offsetParent为$("#first-demo-parent")

    <div id="second-demo-parent" style="position: fixed;">
        <div id="second-demo">xxxxxxxxxxxxxxxxxxxx</div>
    </div>
    $("#second-demo").offsetParent为$("#second-demo-parent")

    <div id="third-demo-parent" style="position: absolute;">
        <div id="third-demo">xxxxxxxxxxxxxxxxxxxx</div>
    </div>
    $("#third-demo").offsetParent为$("#third-demo-parent")

    <div id="four-demo-parent">
        <div id="four-demo">xxxxxxxxxxxxxxxxxxxx</div>
    </div>
    $("#four-demo").offsetParent为$("body")
